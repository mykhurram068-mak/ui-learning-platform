<!-- HTML Structure -->
<div class="form-card">
  <div class="form-tabs">
    <button class="tab-btn active">Login</button>

  </div>
 
  <!-- Login Form -->
  <form id="loginForm">
    <input type="email" placeholder="Email">
    <input type="password" placeholder="Password">
    <button type="submit">Login</button>
  </form>
 
  <!-- Signup Form -->
  <form id="signupForm">
    <input type="text" placeholder="Full Name">

  </form>
</div>

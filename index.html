    // Ensure the demo navbar works correctly inside the page
    (function fixDemoNavbar() {
      const hamburger = document.querySelector('.hamburger-demo');
      const menu = document.querySelector('.nav-menu-demo');
      if (hamburger && menu) {
        // remove any previous listeners and add fresh
        const newHamburger = hamburger.cloneNode(true);
        hamburger.parentNode.replaceChild(newHamburger, hamburger);
        newHamburger.addEventListener('click', (e) => {
          e.stopPropagation();
          menu.classList.toggle('active');
        });
        document.querySelectorAll('.nav-menu-demo a').forEach(link => {
          link.addEventListener('click', () => menu.classList.remove('active'));
        });
      }
    })();
